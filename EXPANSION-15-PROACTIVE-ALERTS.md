## CONTEXT — WHAT AND WHY

Instead of waiting for reps to ask, the bot proactively sends DMs with important pipeline alerts: stale deals, approaching close dates, competitor mentions, and pipeline health changes.

This shifts the bot from reactive to proactive — the single biggest differentiator for executive demos. It makes the AI feel like a real team member who's always watching the pipeline.

**Important:** For the POC, this runs on-demand via `/alerts` AND on a background schedule (using a simple timer loop in Python). In production, you'd use a proper scheduler (APScheduler, Celery, or a cron job).

---

## WHAT IT DOES

1. **On-demand:** Rep types `/alerts` to see their current notifications
2. **Scheduled:** Bot checks every N minutes (configurable) and sends DMs for:
   - Deals with no activity in 14+ days
   - Deals past their close date
   - Deals closing within 3 days with no recent activity
   - New competitor mentions in deal notes
3. Each alert has action buttons to view the deal or take action

```python
"""Block Kit formatting for proactive alerts."""
from formatters.helpers import timestamp, format_currency


def format_alerts_digest(
    alerts: list,
    total_pipeline_value: float,
) -> tuple[list, str]:
    """Format a digest of all current alerts as Block Kit.

    Args:
        alerts: List of dicts with 'type', 'emoji', 'title', 'detail', 'deal_name', 'amount'
        total_pipeline_value: Total value of open pipeline

    Returns:
        tuple: (blocks, fallback_text)
    """
    if not alerts:
        blocks = [
            {
                "type": "section",
                "text": {"type": "mrkdwn", "text": "✅ *All clear!* No pipeline alerts right now. Keep up the great work."}
            },
            {
                "type": "context",
                "elements": [
                    {"type": "mrkdwn", "text": f"Pipeline: {format_currency(total_pipeline_value)} • {timestamp()}"}
                ]
            },
        ]
        return blocks, "No pipeline alerts."

    blocks = [
        {
            "type": "header",
            "text": {"type": "plain_text", "text": f"🔔 Pipeline Alerts ({len(alerts)})", "emoji": True}
        },
        {
            "type": "context",
            "elements": [
                {"type": "mrkdwn", "text": f"Pipeline: {format_currency(total_pipeline_value)} • {timestamp()}"}
            ]
        },

        {"type": "divider"},
    ]

    for i, alert in enumerate(alerts[:8]):
        emoji = alert.get("emoji", "⚠️")
        title = alert.get("title", "Alert")
        detail = alert.get("detail", "")
        deal_name = alert.get("deal_name", "")
        amount = alert.get("amount", 0)

        blocks.append({
            "type": "section",
            "text": {
                "type": "mrkdwn",
                "text": f"{emoji} *{title}*\n{detail}\n_{format_currency(amount)}_"
            },
            "accessory": {
                "type": "button",
                "text": {"type": "plain_text", "text": "View Deal", "emoji": True},
                "action_id": f"alert_view_deal",
                "value": deal_name,
            }
        })

        if i < len(alerts) - 1:
            blocks.append({"type": "divider"})

    blocks.append({"type": "divider"})
    blocks.append({
        "type": "actions",
        "elements": [
            {
                "type": "button",
                "text": {"type": "plain_text", "text": "📈 View Pipeline", "emoji": True},
                "action_id": "refresh_pipeline",
                "value": "refresh",
            },
            {
                "type": "button",
                "text": {"type": "plain_text", "text": "📬 View Follow-Ups", "emoji": True},
                "action_id": "home_followups",
                "value": "followups",
            },
        ]
    })

    blocks.append({
        "type": "context",
        "elements": [
            {"type": "mrkdwn", "text": f"Generated by SalesCoach AI • Proactive Alerts • {timestamp()}"}
        ]
    })

    fallback = f"Pipeline Alerts: {len(alerts)} items need attention"
    return blocks, fallback


def format_single_alert(
    emoji: str,
    title: str,
    detail: str,
    deal_name: str,
    amount: float,
) -> tuple[list, str]:
    """Format a single proactive alert for DM delivery.

    Returns:
        tuple: (blocks, fallback_text)
    """
    blocks = [
        {
            "type": "section",
            "text": {
                "type": "mrkdwn",
                "text": f"{emoji} *{title}*\n{detail}"
            },
            "accessory": {
                "type": "button",
                "text": {"type": "plain_text", "text": "Take Action", "emoji": True},
                "action_id": "alert_view_deal",
                "value": deal_name,
            }
        },
        {
            "type": "context",
            "elements": [
                {"type": "mrkdwn", "text": f"SalesCoach AI • {timestamp()}"}
            ]
        },
    ]

    fallback = f"{emoji} {title}: {detail}"
    return blocks, fallback
```

## STEP 3: Add `/alerts` Command Handler to `app.py`

```python
@app.command("/alerts")
def handle_alerts(ack, say, command):
    """Show current pipeline alerts."""
    ack()
    channel_id = command["channel_id"]

    # Build alerts list
    alerts = build_pipeline_alerts()

    # Total pipeline value
    open_deals = mock_crm.get_all_open_deals()
    total_value = sum(float(d.get("properties", {}).get("amount", 0) or 0) for d in open_deals)

    from formatters import format_alerts_digest
    blocks, fallback = format_alerts_digest(alerts, total_value)

    say(text=fallback, blocks=blocks, channel=channel_id)


def build_pipeline_alerts() -> list:
    """Scan the pipeline and build a list of alerts."""
    from datetime import datetime, timedelta

    alerts = []

    # Stale deals
    stale = mock_crm.get_stale_deals(days=14)
    for deal in stale:
        props = deal.get("properties", {})
        alerts.append({
            "type": "stale",
            "emoji": "⏰",
            "title": f"Stale: {props.get('dealname', 'Unknown')}",
            "detail": "No activity in 14+ days. Time to re-engage.",
            "deal_name": props.get("dealname", ""),
            "amount": float(props.get("amount", 0) or 0),
        })

    # Overdue deals
    overdue = mock_crm.get_overdue_deals()
    for deal in overdue:
        props = deal.get("properties", {})
        # Avoid duplicates with stale
        deal_name = props.get("dealname", "")
        if not any(a["deal_name"] == deal_name for a in alerts):
            alerts.append({
                "type": "overdue",
                "emoji": "🚨",
                "title": f"Overdue: {deal_name}",
                "detail": f"Past close date ({props.get('closedate', 'N/A')[:10]}). Update the timeline or close.",
                "deal_name": deal_name,
                "amount": float(props.get("amount", 0) or 0),
            })

    # Deals closing soon (within 3 days)
    open_deals = mock_crm.get_all_open_deals()
    for deal in open_deals:
        props = deal.get("properties", {})
        close_date = props.get("closedate", "")
        deal_name = props.get("dealname", "")

        if close_date and deal_name not in [a["deal_name"] for a in alerts]:
            try:
                close = datetime.fromisoformat(close_date.replace("Z", "+00:00")).replace(tzinfo=None)
                days_until = (close - datetime.now()).days
                if 0 < days_until <= 3:
                    alerts.append({
                        "type": "closing_soon",
                        "emoji": "⚡",
                        "title": f"Closing Soon: {deal_name}",
                        "detail": f"Closes in {days_until} day{'s' if days_until != 1 else ''}. Confirm next steps.",
                        "deal_name": deal_name,
                        "amount": float(props.get("amount", 0) or 0),
                    })
            except (ValueError, TypeError):
                pass

    # Competitor mentions in recent notes
    for deal in open_deals:
        props = deal.get("properties", {})
        deal_name = props.get("dealname", "")
        for activity in deal.get("activities", []):
            body = activity.get("body", activity.get("note", "")).lower()
            if any(comp in body for comp in ["competitoralpha", "competitorbeta", "competitorgamma", "competitor"]):
                if deal_name not in [a["deal_name"] for a in alerts if a["type"] == "competitor"]:
                    alerts.append({
                        "type": "competitor",
                        "emoji": "⚔️",
                        "title": f"Competitor Alert: {deal_name}",
                        "detail": "Competitor mentioned in recent activity. Review battle cards.",
                        "deal_name": deal_name,
                        "amount": float(props.get("amount", 0) or 0),
                    })
                break

    # Sort by priority: overdue > closing_soon > stale > competitor
    priority = {"overdue": 0, "closing_soon": 1, "stale": 2, "competitor": 3}
    alerts.sort(key=lambda a: priority.get(a["type"], 99))

    return alerts
```
