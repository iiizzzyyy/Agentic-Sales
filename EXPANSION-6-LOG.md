# Claude Code Instructions: `/log` — Quick Activity Logging

> **Read `CLAUDE-CODE-BRIEFING.md`, `app.py`, `SEED-HUBSPOT-INSTRUCTIONS.md` (for HubSpot Python SDK patterns), and `tools/rag.py` first.**

---

## CONTEXT — WHAT AND WHY

Reps hate switching to HubSpot just to log a call note. `/log` lets them record activity directly from Slack in 5 seconds. The AI enhances their raw notes — expanding shorthand, adding structure, and tagging the right deal.

For the POC, this logs to the in-memory mock CRM and optionally writes back to HubSpot via the Python SDK (since the MCP server is read-only). For demo purposes, even just showing the formatted note in Slack is impressive.

---

## WHAT IT DOES

1. Rep types `/log NovaTech — called Sarah, discussed pricing, she wants 15% discount, will send proposal tomorrow`
2. Bot parses:
   - Company/deal from the text
   - Raw notes
3. LLM enhances the note: expands abbreviations, adds structure, extracts action items
4. Bot posts a formatted activity log with:
   - Enhanced note
   - Extracted action items
   - Deal it's attached to
   - "Save to HubSpot" button (if SDK is available)

```python
"""Block Kit formatting for /log command."""
import json
from formatters.helpers import timestamp, truncate


def format_activity_log(
    company_name: str,
    deal_name: str,
    activity_type: str,
    raw_note: str,
    enhanced_note: str,
    action_items: list,
    saved_to_crm: bool = False,
) -> tuple[list, str]:
    """Format an activity log entry as Block Kit.

    Args:
        company_name: Company name
        deal_name: Associated deal name
        activity_type: "call" | "meeting" | "email" | "note"
        raw_note: Original text from the rep
        enhanced_note: LLM-enhanced version
        action_items: List of extracted action item strings
        saved_to_crm: Whether the note was saved to HubSpot

    Returns:
        tuple: (blocks, fallback_text)
    """
    type_emoji = {
        "call": "📞",
        "meeting": "🤝",
        "email": "✉️",
        "note": "📝",
    }
    emoji = type_emoji.get(activity_type, "📝")

    # Action items
    if action_items:
        actions_str = "\n".join([f"• ☐ {item}" for item in action_items[:5]])
    else:
        actions_str = "_No action items extracted_"

    # CRM status
    if saved_to_crm:
        crm_status = "✅ Saved to HubSpot"
    else:
        crm_status = "💾 Not yet saved to CRM"

    blocks = [
        {
            "type": "header",
            "text": {"type": "plain_text", "text": f"{emoji} Activity Logged", "emoji": True}
        },
        {
            "type": "context",
            "elements": [
                {"type": "mrkdwn", "text": f"*{company_name}* • {deal_name} • {crm_status}"}
            ]
        },

        {"type": "divider"},

        # Enhanced note
        {
            "type": "section",
            "text": {"type": "mrkdwn", "text": f"*Note:*\n{enhanced_note}"}
        },

        {"type": "divider"},

        # Action items
        {
            "type": "section",
            "text": {"type": "mrkdwn", "text": f"*Action Items:*\n{actions_str}"}
        },

        # Original note (collapsed)
        {
            "type": "context",
            "elements": [
                {"type": "mrkdwn", "text": f"_Original: {truncate(raw_note, 200)}_"}
            ]
        },

        # Action buttons
        {
            "type": "actions",
            "elements": [
                {
                    "type": "button",
                    "text": {"type": "plain_text", "text": "Save to HubSpot", "emoji": True},
                    "action_id": "save_note_to_hubspot",
                    "value": json.dumps({
                        "company": company_name,
                        "deal": deal_name,
                        "note": enhanced_note,
                    }),
                    "style": "primary",
                },
                {
                    "type": "button",
                    "text": {"type": "plain_text", "text": "Edit Note", "emoji": True},
                    "action_id": "edit_activity_note",
                    "value": json.dumps({
                        "company": company_name,
                        "deal": deal_name,
                        "note": enhanced_note,
                    }),
                },
                {
                    "type": "button",
                    "text": {"type": "plain_text", "text": "Draft Follow-Up", "emoji": True},
                    "action_id": "draft_email_from_deal",
                    "value": company_name,
                },
            ]
        },

        {
            "type": "context",
            "elements": [
                {"type": "mrkdwn", "text": f"Generated by SalesCoach AI • {timestamp()}"}
            ]
        },
    ]

    fallback = f"Activity logged: {company_name} — {activity_type}"
    return blocks, fallback
```

## STEP 3: Add `/log` Command Handler to `app.py`

```python
@app.command("/log")
def handle_log(ack, say, command):
    """Quick activity logging with AI enhancement."""
    ack()
    raw_text = command["text"].strip()
    channel_id = command["channel_id"]
    user_id = command["user_id"]

    if not raw_text:
        say(
            "Log an activity: `/log [company] — [your notes]`\n"
            "Examples:\n"
            "• `/log NovaTech — called Sarah, discussed pricing, she wants 15% discount`\n"
            "• `/log Apex — demo went well, CTO asked about security, sending SOC2`\n"
            "• `/log Greenfield — meeting canceled, rescheduling for next week`",
            channel=channel_id,
        )
        return

    say(f"📝 Logging activity...", channel=channel_id)

    # 1. Parse company name from text (split on — or -)
    company_name = ""
    raw_note = raw_text

    if "—" in raw_text:
        parts = raw_text.split("—", 1)
        company_name = parts[0].strip()
        raw_note = parts[1].strip()
    elif " - " in raw_text:
        parts = raw_text.split(" - ", 1)
        company_name = parts[0].strip()
        raw_note = parts[1].strip()

    # 2. Find associated deal
    deal_name = "N/A"
    if company_name:
        company_data = mock_crm.search_company(company_name)
        if company_data:
            deals = mock_crm.get_deals_for_company(company_data.get("id", ""))
            if deals:
                deal_name = deals[0].get("properties", {}).get("dealname", "N/A")

    # 3. Detect activity type
    activity_type = "note"
    text_lower = raw_note.lower()
    if any(w in text_lower for w in ["called", "call", "phone", "rang"]):
        activity_type = "call"
    elif any(w in text_lower for w in ["meeting", "met", "met with", "demo"]):
        activity_type = "meeting"
    elif any(w in text_lower for w in ["emailed", "sent email", "replied"]):
        activity_type = "email"

    # 4. Run LLM to enhance the note
    result = app_graph.invoke({
        "messages": [HumanMessage(content=f"""A sales rep just logged this quick note about {company_name}:

"{raw_note}"

Do TWO things:

1. ENHANCED NOTE: Rewrite this as a professional CRM activity note. Expand abbreviations, add structure, keep it concise (3-5 sentences max). Don't add information that isn't in the original — just clean it up.

2. ACTION ITEMS: Extract specific next steps from the note. Each action item should be a clear task.

Format:
ENHANCED_NOTE: [the enhanced note]

ACTION_ITEMS:
- [action item 1]
- [action item 2]
- [action item 3]""")],
        "route": "assistant",
        "workflow": "research",
        "user_id": user_id,
        "channel_id": channel_id,
        "thread_ts": None,
    })

    llm_response = result["messages"][-1].content

    # 5. Parse LLM response
    enhanced_note = raw_note  # Fallback
    action_items = []

    lines = llm_response.split("\n")
    current_section = ""
    for line in lines:
        line_stripped = line.strip()
        if line_stripped.upper().startswith("ENHANCED_NOTE:"):
            enhanced_note = line_stripped.split(":", 1)[1].strip()
            current_section = "note"
        elif "ACTION_ITEM" in line_stripped.upper() or "ACTION ITEM" in line_stripped.upper():
            current_section = "actions"
        elif current_section == "note" and line_stripped and not line_stripped.startswith("-"):
            enhanced_note += " " + line_stripped
        elif current_section == "actions":
            cleaned = line_stripped.lstrip("•-* 0123456789.")
            if cleaned:
                action_items.append(cleaned)

    # 6. Format and send
    from formatters import format_activity_log
    blocks, fallback = format_activity_log(
        company_name=company_name or "Unknown Company",
        deal_name=deal_name,
        activity_type=activity_type,
        raw_note=raw_note,
        enhanced_note=enhanced_note.strip(),
        action_items=action_items,
        saved_to_crm=False,
    )

    say(text=fallback, blocks=blocks, channel=channel_id)
```

        hs_client = HubSpot(access_token=token)

        # Create the note
        from hubspot.crm.objects.notes import SimplePublicObjectInputForCreate as NoteInput
        from datetime import datetime

        result = hs_client.crm.objects.notes.basic_api.create(
            simple_public_object_input_for_create=NoteInput(
                properties={
                    "hs_timestamp": datetime.now().strftime("%Y-%m-%dT%H:%M:%S.000Z"),
                    "hs_note_body": f"[Logged via SalesCoach AI]\n\n{note_text}",
                }
            )
        )

        client.chat_postMessage(
            channel=channel,
            text=f"✅ Note saved to HubSpot (ID: {result.id})",
        )

    except ImportError:
        client.chat_postMessage(
            channel=channel,
            text="⚠️ HubSpot Python SDK not installed. Run `pip install hubspot-api-client` to enable CRM write-back.",
        )
    except Exception as e:
        client.chat_postMessage(
            channel=channel,
            text=f"⚠️ Could not save to HubSpot: {str(e)[:100]}. The note is displayed above — you can copy it manually.",
        )


@app.action("edit_activity_note")
def handle_edit_activity_note(ack, body, client):
    """Open a modal to edit the activity note before saving."""
    ack()
    action_data = json.loads(body["actions"][0]["value"])
    trigger_id = body["trigger_id"]

    client.views_open(
        trigger_id=trigger_id,
        view={
            "type": "modal",
            "callback_id": "submit_edited_note",
            "title": {"type": "plain_text", "text": "Edit Activity Note"},
            "submit": {"type": "plain_text", "text": "Save"},
            "blocks": [
                {
                    "type": "input",
                    "block_id": "note_block",
                    "element": {
                        "type": "plain_text_input",
                        "action_id": "note_input",
                        "multiline": True,
                        "initial_value": action_data.get("note", ""),
                    },
                    "label": {"type": "plain_text", "text": "Activity Note"},
                },
            ],
            "private_metadata": json.dumps({
                "company": action_data.get("company", ""),
                "deal": action_data.get("deal", ""),
            }),
        }
    )


@app.view("submit_edited_note")
def handle_submit_edited_note(ack, body, client):
    """Handle the edited note modal submission."""
    ack()
    values = body["view"]["state"]["values"]
    edited_note = values["note_block"]["note_input"]["value"]
    metadata = json.loads(body["view"].get("private_metadata", "{}"))
    user_id = body["user"]["id"]

    # Send confirmation DM
    client.chat_postMessage(
        channel=user_id,
        text=f"📝 Updated note for *{metadata.get('company', 'Unknown')}*:\n\n{edited_note}\n\nUse the Save to HubSpot button to push this to CRM.",
    )
